#include<bits/stdc++.h>

#define int int64_t

using namespace std;

int32_t main() {
    ios_base::sync_with_stdio(false); cin.tie(NULL);
    
    string n;

    cin >> n;

    vector<int32_t> a(10, 0), b(10, 0), zero;
    vector<int32_t> x, y;
    int d = n.size();

    for(auto i : n) {
        a[i - '0']++;
        b[i - '0']++;
    }

    bool rem = 0;
    while(d--) {
        bool flag = 0;
        if(rem) {
            for(int i = 1; i <= 9; i++) 
                if(a[i] && b[9 - i]) {
                    rem = 0;
                    x.push_back(i);
                    y.push_back(9 - i);
                    a[i]--;
                    b[9 - i]--;
                    flag = 1;
                    break;
                }
        } 
        else {
            for(int i = 1; i <= 9; i++) 
                if(a[i] && b[10 - i]) {
                    rem = 1;
                    x.push_back(i);
                    y.push_back(10 - i);
                    a[i]--;
                    b[10 - i]--;
                    flag = 1;
                    break;
                }
        }
        if(!flag) {
            while(a[0] && b[0]) {
                a[0]--, b[0]--;
                zero.push_back(0);                
            }
            // if(rem) {
            //     x.push_back(1);
            //     y.push_back(1);
            // }
            for(int i = 0; i <= 9; i++) {
                while(a[i]) {
                    x.push_back(i);
                    a[i]--;
                }
                while(b[i]) {
                    y.push_back(i); 
                    b[i]--;
                }
            }
            break;
        }        
    }

    reverse(x.begin(), x.end());
    reverse(y.begin(), y.end());

    for(auto i : x) cout << i; 
    if(!zero.empty()) for(auto j : zero) cout << j;
    cout << '\n';
    for(auto i : y) cout << i;
    if(!zero.empty()) for(auto j : zero) cout << j;
}
